这是一个综合展示 `Debug`、`PartialEq` 和 `Default` 三种常用自动派生功能的代码示例：

```rust
// 1. Debug: 允许使用 {:?} 打印结构体以便调试
// 2. PartialEq: 允许使用 == 和 != 运算符比较成员
// 3. Default: 为类型提供默认构造值 (String:"", u32:0, bool:false)
#[derive(Debug, PartialEq, Default)]
struct Player {
    name: String,
    level: u32,
    is_active: bool,
}

fn main() {
    // 【用法 1: Default】
    // 自动生成初始值，无需手动写构造函数
    let default_player = Player::default();

    // 【用法 2: Debug】
    // 直接打印整个结构体的内容
    println!("默认玩家状态: {:?}", default_player);
    // 输出: Player { name: "", level: 0, is_active: false }

    let p1 = Player {
        name: String::from("Rustacean"),
        level: 99,
        is_active: true,
    };

    let p2 = Player {
        name: String::from("Rustacean"),
        level: 99,
        is_active: true,
    };

    // 【用法 3: PartialEq】
    // 逐字段比较两个结构体是否相等
    if p1 == p2 {
        println!("两个玩家的数据完全一致！");
    }
}
```
